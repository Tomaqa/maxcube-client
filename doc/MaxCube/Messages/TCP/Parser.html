def parse_tcp_data(raw_data)
  check_tcp_data(raw_data)
  raw_data.split("\r\n").map(&method(:parse_tcp_msg))
end

def parse_tcp_msg(msg)
  check_tcp_msg(msg)
  body = msg.split(':')[1] || ''
  hash = { type: @msg_type }
  return hash unless parse_msg_body(body, hash, 'tcp')
  check_tcp_hash(hash)
end
