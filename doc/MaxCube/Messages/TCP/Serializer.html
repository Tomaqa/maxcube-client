def serialize_tcp_hash(hash)
  check_tcp_hash(hash)
  msg = "#{@msg_type}:" << serialize_hash_body(hash, 'tcp')
  check_tcp_msg(msg) << "\r\n"
end

def serialize_tcp_hashes(hashes)

  check_tcp_data(raw_data)
